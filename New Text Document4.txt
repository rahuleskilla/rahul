pragma solidity 0.8.0; 
contract division {
    uint256 firstNo;
    uint256 secondNo;
function firstNoSet(uint256 x) public {
    firstNo = x;

}
function SecondNoSet(uint256 y) public {
    secondNo = y;
}
function divi () public view returns (uint256){
    uint256 Div = firstNo/secondNo;
    return Div;
}
}
